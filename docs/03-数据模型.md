# 数据模型

## 1. 数据库设计

### 1.1 数据库表概览

| 表名 | 说明 | 主要字段 |
|------|------|---------|
| `resumes` | 简历基础信息 | id, file_hash, original_filename, storage_key, resume_text |
| `resume_analyses` | 简历分析结果 | id, resume_id, overall_score, strengths_json, suggestions_json |
| `interview_sessions` | 面试会话 | id, session_id, resume_id, status, questions_json |
| `interview_answers` | 面试答案 | id, session_id, question_index, user_answer, score |
| `knowledge_bases` | 知识库文档 | id, file_hash, name, category, vector_status |
| `rag_chat_sessions` | RAG 聊天会话 | id, title, status, message_count |
| `rag_chat_messages` | RAG 聊天消息 | id, session_id, type, content, message_order |

## 2. 实体模型

### 2.1 简历模块

#### ResumeEntity

```java
@Entity
@Table(name = "resumes", indexes = {
    @Index(name = "idx_resume_hash", columnList = "file_hash", unique = true)
})
public class ResumeEntity {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "file_hash", unique = true, nullable = false, length = 64)
    private String fileHash;

    @Column(name = "original_filename", nullable = false, length = 255)
    private String originalFilename;

    @Column(name = "file_size")
    private Long fileSize;

    @Column(name = "content_type", length = 100)
    private String contentType;

    @Column(name = "storage_key", length = 500)
    private String storageKey;

    @Column(name = "storage_url", length = 500)
    private String storageUrl;

    @Column(name = "resume_text", columnDefinition = "TEXT")
    private String resumeText;

    @Column(name = "uploaded_at", nullable = false)
    private LocalDateTime uploadedAt;

    @Column(name = "last_accessed_at")
    private LocalDateTime lastAccessedAt;

    @Column(name = "access_count")
    private Integer accessCount = 0;

    @Enumerated(EnumType.STRING)
    @Column(name = "analyze_status")
    private AsyncTaskStatus analyzeStatus = AsyncTaskStatus.PENDING;

    @Column(name = "analyze_error", columnDefinition = "TEXT")
    private String analyzeError;

    // 关联关系
    @OneToMany(mappedBy = "resume", cascade = CascadeType.ALL)
    private List<ResumeAnalysisEntity> analyses;

    @OneToMany(mappedBy = "resume", cascade = CascadeType.ALL)
    private List<InterviewSessionEntity> interviewSessions;
}
```

#### ResumeAnalysisEntity

```java
@Entity
@Table(name = "resume_analyses")
public class ResumeAnalysisEntity {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "resume_id")
    private ResumeEntity resume;

    // 总分 0-100
    @Column(name = "overall_score")
    private Integer overallScore;

    // 各维度评分
    @Column(name = "content_score")
    private Integer contentScore;       // 0-25

    @Column(name = "structure_score")
    private Integer structureScore;     // 0-20

    @Column(name = "skill_match_score")
    private Integer skillMatchScore;    // 0-25

    @Column(name = "expression_score")
    private Integer expressionScore;    // 0-15

    @Column(name = "project_score")
    private Integer projectScore;       // 0-15

    @Column(name = "summary", columnDefinition = "TEXT")
    private String summary;

    @Column(name = "strengths_json", columnDefinition = "TEXT")
    private String strengthsJson;

    @Column(name = "suggestions_json", columnDefinition = "TEXT")
    private String suggestionsJson;

    @Column(name = "analyzed_at")
    private LocalDateTime analyzedAt;
}
```

### 2.2 面试模块

#### InterviewSessionEntity

```java
@Entity
@Table(name = "interview_sessions", indexes = {
    @Index(name = "idx_interview_session_resume_created", columnList = "resume_id,created_at"),
    @Index(name = "idx_interview_session_resume_status_created", columnList = "resume_id,status,created_at")
})
public class InterviewSessionEntity {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "session_id", unique = true, nullable = false, length = 64)
    private String sessionId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "resume_id")
    private ResumeEntity resume;

    @Column(name = "total_questions")
    private Integer totalQuestions;

    @Column(name = "current_question_index")
    private Integer currentQuestionIndex = 0;

    @Enumerated(EnumType.STRING)
    @Column(name = "status")
    private SessionStatus status = SessionStatus.CREATED;

    @Column(name = "questions_json", columnDefinition = "TEXT")
    private String questionsJson;

    @Column(name = "overall_score")
    private Integer overallScore;

    @Column(name = "overall_feedback", columnDefinition = "TEXT")
    private String overallFeedback;

    @Column(name = "strengths_json", columnDefinition = "TEXT")
    private String strengthsJson;

    @Column(name = "improvements_json", columnDefinition = "TEXT")
    private String improvementsJson;

    @Column(name = "reference_answers_json", columnDefinition = "TEXT")
    private String referenceAnswersJson;

    @OneToMany(mappedBy = "session", cascade = CascadeType.ALL)
    private List<InterviewAnswerEntity> answers;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @Column(name = "completed_at")
    private LocalDateTime completedAt;

    @Enumerated(EnumType.STRING)
    @Column(name = "evaluate_status")
    private AsyncTaskStatus evaluateStatus;

    @Column(name = "evaluate_error", columnDefinition = "TEXT")
    private String evaluateError;

    enum SessionStatus {
        CREATED,      // 已创建
        IN_PROGRESS,  // 进行中
        COMPLETED,    // 已完成
        EVALUATED     // 已评估
    }
}
```

#### RagChatSessionEntity

```java
@Entity
@Table(name = "rag_chat_sessions", indexes = @Index(name = "idx_rag_session_updated", columnList = "updated_at"))
public class RagChatSessionEntity {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "title", length = 255)
    private String title;

    @Enumerated(EnumType.STRING)
    @Column(name = "status")
    private SessionStatus status = SessionStatus.ACTIVE;

    @ManyToMany
    @JoinTable(
        name = "rag_session_kb_mapping",
        joinColumns = @JoinColumn(name = "session_id"),
        inverseJoinColumns = @JoinColumn(name = "kb_id")
    )
    private Set<KnowledgeBaseEntity> knowledgeBases;

    @OneToMany(mappedBy = "session", cascade = CascadeType.ALL)
    @OrderBy("messageOrder ASC")
    private List<RagChatMessageEntity> messages;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    @Column(name = "message_count")
    private Integer messageCount = 0;

    @Column(name = "is_pinned")
    private Boolean isPinned = false;

    enum SessionStatus {
        ACTIVE,    // 活跃
        ARCHIVED   // 已归档
    }
}
```
